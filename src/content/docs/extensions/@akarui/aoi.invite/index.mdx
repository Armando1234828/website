To use aoi.invite, you are required to edit your index.js file.

```javascript
const { AoiClient } = require("aoi.js");
const { InviteManager } = require("@akarui/aoi.invite");

const client = new AoiClient({
  token: "Discord Bot Token",
  prefix: "Discord Bot Prefix",
  intents: ["MessageContent", "Guilds", "GuildMessages"],
  events: ["onMessage", "onInteractionCreate"],
  database: {
    type: "aoi.db",
    db: require("@akarui/aoi.db"),
    dbType: "KeyValue",
    tables: ["main"],
    securityKey: "a-32-characters-long-string-here",
  },
});

// Ping Command
client.command({
  name: "ping",
  code: `Pong! $pingms`,
});

new InviteManager(
  client,
  {
    sk: "a-32-characters-long-string-here",
  },
  ["inviteJoin", "inviteLeave"]
);
```

---

### InviteEvents

- inviteJoin (triggered on join)
- inviteLeave (triggered on leave)
- inviteError (triggered on error)
